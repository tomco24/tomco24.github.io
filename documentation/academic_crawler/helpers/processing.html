import pprint
from collections import defaultdict, Counter

import pandas as pd

from academic_crawler.config.state import ProgramState
from academic_crawler.scrapers.templates import Organization, Person, Affiliation
import requests
import json

def process_scraped_org_structure_items(org_structure_items) -> list[Organization]:
    def get_status_code(url):
        try:
            return requests.get(url, timeout=5).status_code
        except:
            return 404
    def filter_orgs(instances:list[Organization])->list[Organization]:
        result = []
        domain_dict = defaultdict(list)
        for instance in instances:
            if instance.parent.is_subdomain(instance.url):
                domain_dict[instance.domain].append(instance)
            else:
                result.append(instance)

        for domain,instances in domain_dict.items():
            if len(instances) == 1:
                result.append(instances[0])
            else:
                most_common_name = Counter([instance.name for instance in instances]).most_common(1)[0][0]
                for instance in instances:
                    if instance.name == most_common_name:
                        result.append(instance)
                        break

        return result


    result = []
    for item in org_structure_items:
        if item['type'] == 'ORGANIZATION':
            status_code = get_status_code(item['instance'].url)
            print(f"Status code for {item['instance'].url} is {status_code}")
            if status_code == 200:
                result.append(item['instance'])
    result = filter_orgs(result)
    pprint.pprint(result)
    return result

def process_scraped_people_items(people_items) -> list[Person]:
    result = []
    for item in people_items:
        person = item.get('person', None)
        if person is None or len(person.name) == 0 or len(person.name) > 50 or (not isinstance(person, Person)):
            continue
        person.add_implicit_affiliations()
        result.append(person)
    return result


class ReferenceRawProcessor:
    def __init__(self) -> None:
        super().__init__()

    def process_root_organization(self, df: pd.DataFrame, cols=('university_1', 'division'), meta=None):
        df['organization'] = meta
        root_col = cols[0]
        col_1_is_none = df[cols[0]].isna().all()
        if col_1_is_none:
            col_2_is_none = df[cols[1]].isna().all()
            if col_2_is_none:
                return None
            else:
                # drop column cols[0]
                # df.drop(columns=cols[0],inplace=True)
                root_col = cols[1]
                return root_col
        # set values for root_col to meta.name if  value is None
        df[root_col].fillna(meta.name, inplace=True)
        # if values for root_col != meta.name put these values into separata variable
        separate_vals = df[df[root_col] != meta.name][root_col].unique().tolist()
        if len(separate_vals) > 0:
            sub_orgs = {}
            for value in separate_vals:
                new_meta = Organization(name=value, url="", parent=meta, domain="")
                sub_orgs[value] = new_meta
            df['organization'] = df[root_col].apply(lambda x: sub_orgs[x] if x in sub_orgs else meta)

        # replace value 'organization' column with Organization instance from sub_orgs

        df.dropna(subset=[root_col], inplace=True)
        # remove
        val_count = df[root_col].value_counts().argmax()
        print(f"Root column is {root_col}")
        print(f"Value counts for root column are {val_count}")

        pass

    def process_affiliations(self, df: pd.DataFrame):
        affiliations = []
        df['affiliations'] = None
        # iterate over rows as dicts and create Affiliation instances
        for index, row in df.iterrows():
            affiliations = []
            if row['division'] is not None:
                affiliation = Affiliation(division=row['division'], position=row['role'])
                affiliations.append(affiliation)
            # add it to the row in dataframe
            df.at[index, 'affiliations'] = affiliations
        return affiliations

    def process_people(self, df: pd.DataFrame):
        people = []
        df['person'] = None
        # iterate over rows as dicts and create Person instances
        for index, row in df.iterrows():
            person = Person(organization=row['organization'], name=row['name'])
            person.affiliations = row['affiliations']
            people.append(person)
            df.at[index, 'person'] = person
        return people

    def load_raw(self, path):
        df_raw = pd.read_csv(path)
        columns = ['name', 'university_1', 'division', 'role']
        df = df_raw.reindex(columns=columns, fill_value=None)
        return df
