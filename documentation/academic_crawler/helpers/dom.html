class ElementCategoryHelper:
    def __init__(self):
        self.elements_categories = elements_categories


    def get_category(self,tag):
        general_category = self.get_general_category(tag.name)
        if self.is_textual(general_category):
            return general_category,'text'
        if self.is_main_section(tag,general_category):
            return general_category,'main'
        if self.is_nav_section(tag,general_category):
            return general_category,'nav'
        return general_category,'other'

    def get_general_category(self, tag_name):
        for category, tags in self.elements_categories.items():
            if tag_name in tags:
                return category
        return None

    def is_main_section(self,tag,category:ElementCategory):
        if category == ElementCategory.CONTENT_SECTIONING \
                and (tag.name == 'main' or tag.get('role',None)  == 'main'):
            return True
        return False

    def is_nav_section(self,tag,category:ElementCategory):
        if category == ElementCategory.CONTENT_SECTIONING \
                and (tag.name == 'nav' or tag.get('role',None)  == 'navigation'):
            return True
        return False

    def is_textual(self,category:ElementCategory):
        if category == ElementCategory.TEXT_CONTENT:
            return True
        if category == ElementCategory.INLINE_TEXT_SEMANTICS:
            return True
        if category == ElementCategory.TABLE_CONTENT:
            return True
