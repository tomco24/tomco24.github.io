import enum

from academic_crawler.config.domain import DomainConfiguration,SubsidiaryConfiguration
from transitions import Machine


class States(enum.Enum):
    START = 0
    LINKS_DOWNLOAD = 1
    FILTER_LINKS = 2
    SCRAPE_PERSON_PAGE = 3
    SCRAPE_ALL_LINKS = 4
    DETECT_STRUCTURE = 5
    SCRAPE_PEOPLE_DATA_PAGE = 6




class CrawlingController:
    def __init__(self,config:SubsidiaryConfiguration) -> None:
        super().__init__()
        self.config = config
        self.spider = None
        self.without_callback = {States.START}

    def get_domain(self):
        return self.config.characteristics.get("sub_domain")

    def get_start_url(self):
        if self.config.has_people_page():
            return self.config.people_page_url
        else:
            return self.config.characteristics.get("url")

    def set_transition_data(self,type,data):
        if not isinstance(data,list):
            data = [data]
        result = {
            "type":type,
            "data":data,
            "state":self.state,
        }
        self.spider.transition_data = result


    def state_change(self,event):
        self.spider.state_callback(event)


    def first_page_people_and_single_page_per_person(self,event):
        return self.config.has_people_page() and (not self.config.is_single_page)

    def first_page_people_and_people_page_data(self,event):
        return self.config.has_people_page() and self.config.is_single_page

    def get_file_name(self):
        return self.config.characteristics.get("name")


transitions = []

def build_state_machine(config={}):
    controller = CrawlingController(config)
    machine = Machine(controller, States, send_event=True,transitions=transitions, initial=States.START) #after_state_change="state_change"
    machine.add_transition("move",States.START,States.LINKS_DOWNLOAD,conditions=['first_page_people_and_single_page_per_person'])
    machine.add_transition("move",States.LINKS_DOWNLOAD,States.SCRAPE_PERSON_PAGE)
    machine.add_transition("move",States.START,States.DETECT_STRUCTURE,conditions=['first_page_people_and_people_page_data'])

    #machine.add_transition("move",States.FILTER_LINKS,States.SCRAPE_PERSON_PAGE)



    return machine,controller
