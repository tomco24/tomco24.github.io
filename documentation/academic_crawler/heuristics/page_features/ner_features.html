import re

import networkx as nx
import bs4

from ...config.utils import clean_text, find_with_regex,split_string


class NERPageFeatureExtractor(AbstractPageFeatureExtractor):
    ner_detector: NERDetector
    nodes_map: defaultdict[Any, set[SemanticNode]]
    tagged_nodes: list[SemanticNode]
    base_tag_set = {'h1', 'h2', 'h3', 'h4', 'h5', 'h6','a','li','td','tr','button'}
    def __init__(self,state,ner_type) -> None:
        super().__init__(state)
        self.ner_type = ner_type
        self.ner_detector = NERDetector(ner_type)
        self.ner_stats = {key:0 for key in ner_type}
        self.ner_nodes_stats = {key:0 for key in ner_type}
        self.nodes_map = defaultdict(set)
        self.tagged_nodes = []
        self.name_in_heading = False
        self.title_ner= {}

    def extract_features(self):
        self.tree_ner_search()
        self.extract_title_features()
        print(self.ner_stats)
        has_names_in_links, has_siblings_in_links_names = self._ner_in_links("PERSON")
        has_org_in_links, has_siblings_in_links_org = self._ner_in_links("ORG")
        return {
            "num_names_page":self.ner_stats['PERSON'],
            "num_name_nodes":self.ner_nodes_stats['PERSON'],
            "num_ners_page":sum(list(self.ner_stats.values())),
            "has_names_in_links":has_names_in_links,
            "has_siblings_in_links_name":has_siblings_in_links_names,
            "has_other_ners":len(self.ner_stats) > 1,
            "num_orgs_page": self.ner_stats['ORG'],
            "num_org_nodes": self.ner_nodes_stats['ORG'],
            "has_siblings_in_links_org": has_org_in_links,
            "has_orgs_in_links": has_siblings_in_links_org,
            "has_email": 'EMAIL' in self.ner_stats.keys(),
            "has_name_in_heading": self.name_in_heading,
            "has_name_in_title":len(self.title_ner.get('PERSON',{}))>0,
            "has_org_in_title":len(self.title_ner.get('ORG',{}))>0,


        }

    def tree_ner_search(self):
        #for node in nx.bfs_tree(self.tree,"body_0"):

        for node in self.tree.nodes(data=True):
            data_node = node[1].get('data')
            self.node_ner_search(data_node,NERPageFeatureExtractor.base_tag_set)
            #TODO PARENT NER STATS
            #if self.tree.out_degree(node[0]) ==0:
            #    data_node.update_parent_ner_stats()

        for node in self.tree.nodes(data=True):
            data_node = node[1].get('data')
            node_entity = data_node.entity
            if node_entity.ner_tag:
                self.ner_nodes_stats[node_entity.ner_tag] += self.get_ner_node_weight(
                    node_entity.ner_stats[node_entity.ner_tag])


    def node_ner_search(self, node:SemanticNode, tag_set=None):
        node.entity.setup_ner_structure()
        #if (tag_set and node.label not in tag_set and node.parent and node.parent.label not in tag_set):
        #    return
        if not node.leaf:
            return
        i = 0
        fulltext = []
        intervals_idx = []

        result_ner = self.tag_ner_search(node.entity.values)
        if result_ner:
            node.entity.add_ner(result_ner)
            node.entity.was_ner_tagged = True
            #print(node.label,":",result_ner)
        fulltext = []

        if node.entity.ner_tag:
            self.tagged_nodes.append(node)
            self.check_name_in_heading(node)
            self.update_ner_stats(node)
            self.nodes_map[node.label].add(node)



    def update_ner_stats(self, node:SemanticNode):
        node_entity = node.entity
        for key in node_entity.ner_stats.keys():
            self.ner_stats[key] += node_entity.ner_stats[key]
            if node.parent:
                node.parent.entity.ner_stats[key] +=node_entity.ner_stats[key]
                node.parent.entity.update_ner_tag()
        #self.ner_nodes_stats[node_entity.ner_tag] += self.get_ner_node_weight(node_entity.ner_stats[node_entity.ner_tag])


    def get_ner_node_weight(self,n):
        if n<5:
            return 0
        if n<10:
            return 1
        return 2

    def tag_ner_search(self,text, interval_idx=None):
        #text = clean_text(self.get_tag_only_text(tag))
        #print(text)
        result = []
        if len(text) < 1:
            return None
        for item in text:
            if len(item)>50:
                result.append('')
            else:
                result.append(item)
        #print(text)
        result = self.ner_detector.extract_ner(text,[(0,len(text))])
        return result


    def update_node_ner_status(self, node:SemanticNode):
        result = dict(zip(self.ner_type, [False]*len(self.ner_type)))
        for result_ner in node.entity.entities:
            for key in result_ner.keys():
                result[key] = True
        node.ner_stats = result


    def has_found_enough(self):
        for key in self.ner_type:
            if self.ner_stats[key] > 10:
                return True
        return False

    def _ner_in_links(self,type_):
        count = 0
        if type_ not in self.ner_type:
            return False,False
        has_names = False
        has_siblings = False
        for node in self.tagged_nodes:

            if node.is_link_descendant() and node.entity.ner_tag==type_:
                count += node.entity.ner_stats[type_]
                has_names = True
        return has_names,count>1

    def check_name_in_heading(self, node:SemanticNode):
        if self.name_in_heading is None:
            found = node.get_ner_in_tag('PERSON',"h1,h2,h3,h4,h5,h6")
            if len(found)>0:
                self.name_in_heading = True
            self.name_in_heading = False
        return self.name_in_heading

    def extract_title_features(self):
        title = self.state.get_page('title')
        title = clean_text(title)
        result = self.ner_detector.extract_ner_single(title)
        self.title_ner = result
        return result
