import re
from urllib.parse import urlparse
import academic_crawler.config.utils as utils
from furl import furl


class LinkManager:
    def __init__(self, link, history=None) -> None:
        super().__init__()
        self.link = link
        self.url = furl(link.url)
        self.referer_text = utils.clean_text(link.text)

        self.path = utils.parse_path(self.url.path)
        self.history = history
        self.kw_flag = None
        self.name_flag = None
        self.confidence_dict = None
        self.confidence_flags = {}
        self.url_rating = self.get_url_rating()
        self.page_type = None
        self.page_header = None
        self.response = None

    def __hash__(self) -> int:
        return hash(self.link.url)

    def keyword_entity_match(self,detector,entity):
        result = []
        result_text = detector.extract_keyword(self.referer_text).get(entity,None)
        if result_text:
            result.append(('referer_text',result_text))
        #TODO ADD url keyword_category match
        return result

    def contains_keyword(self,detector,category):
        result_text = detector.extract_possible_keywords(self.referer_text).get(category,tuple())
        if len(result_text)>0:
            return True
        #TODO ADD url keyword_category match
        return False




    def get_referer_text_tokens(self):
        return utils.simple_word_tokenization(self.referer_text)
    def check_if_name_in_text_old(self):
        text = self.link.text
        splits = str(text).split()
        is_upper = [x[0].isupper() for x in splits]
        # print(text, "---",is_upper)
        return sum(is_upper) >= 2  # TODO ADD NAME FLAG

    def check_if_name_in_text(self, ncw):
        result = ncw.get_name_confidence(self.link.text)
        if result['confidence_name'] > 0:
            return True
        else:
            return False

    def check_if_keyword_match(self, kcw):
        keyword_confidence_res = kcw.get_link_confidence(self.link.url)
        if keyword_confidence_res['confidence_keyword'] > 0:
            self.kw_flag = True
            return True
        else:
            self.kw_flag = False
            return False

    def set_confidence_flag(self, values, labels):
        if len(values) != len(labels):
            return False
        for i in range(len(values)):
            self.confidence_flags[labels[i]] = values[i]

    def get_referer_text(self):
        # if len(self.history) == 0:
        # return ""
        # text = self.history[-1].text
        # text = text.lower()
        return self.referer_text

    def should_explore(self):
        if self.confidence_flags['name'] and self.confidence_flags['keyword_category'] is False:
            return True
        return False

    def __repr__(self) -> str:
        return '( ' + self.referer_text + "," + self.link.url + " )"

    def get_url_rating(self):
        url = self.url
        fragment = 1 if url.fragment else 0
        return str(url.path).count('/') + str(url.query).count('&') + fragment

    def is_sub_link(self,other: 'LinkManager'):
        other_url = str(other.url.path)
        if other_url.startswith(str(self.url.path)):
            return True
        return False

class UrlPatternMatcher:

    def __init__(self) -> None:
        super().__init__()
        self.patterns = set([])
        self.has_patterns = False

    @staticmethod
    def get_regex_from_start_url(urls):
        result = []
        for url in urls:
            url_ = furl(url)
            segments = url_.path.segments
            if len(segments) > 0:
                escaped = re.escape(url_.pathstr)
                pattern = r".*{}.*$".format(escaped)
                result.append(pattern)
        return result

    def extract_pattern_url(self, url):
        url = furl(url)
        segments = url.path.segments
        query = url.query.params
        if query:
            pattern = url.remove(args=query)
        else:
            pattern = url.remove(path=segments[-1])
        return pattern

    def add_pattern(self, url):
        pattern = self.extract_pattern_url(url).url
        if pattern not in self.patterns:
            self.patterns.add(pattern)
            self.has_patterns = True
            return True
        return False

    def match_patterns(self, url):
        pattern = self.extract_pattern_url(url).url
        return pattern in self.patterns
