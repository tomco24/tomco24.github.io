def prune_DOM_tree_CNR(G:nx.DiGraph,start):
    def computeCNR(G, node:"SemanticNode"):
        if G.out_degree(node.label) == 0 and node.category[1] == "text":
            node.set_cnr(1, len(node.tag.get_text(strip=True)))

            return node
        elif G.out_degree(node.label) == 0 and node.category[1] != "text":
            node.set_cnr(1, 0)
            return node
        else:
            descendants = 0
            char_length = 0
            for child in G.successors(node.label):
                child_data = get_tree_data_node(G, child)
                computeCNR(G, child_data)
                char_length += child_data.cnr_node.text_length
                descendants += child_data.cnr_node.weight
            print(f"{node.label} has {descendants} descendants and {char_length} chars")
            node.set_cnr(descendants, char_length)
            return node
    def identify_main_content_blocks(G,S,trans_closure):
        blocks  = set([node.label for node in S])
        to_remove = set([])
        for node_a in blocks:
            for node_b in blocks:
                if node_a == node_b or node_b in to_remove:
                    continue
                to_find = tuple([node_a,node_b])
                if node_b in set(nx.descendants(G,node_a)):
                    to_remove.add(node_b)
                    break

        blocks = blocks.difference(set(to_remove))
        old_blocks = set(blocks)
        while True:
            parent_in_blocks = None
            for node in old_blocks:
                parent = next(iter(G.predecessors(node)),None)
                if parent is None:
                    continue
                siblings = set(G.successors(parent)).intersection(blocks)
                if len(siblings) >= 2:
                    parent_in_blocks = parent
                    blocks = blocks.difference(siblings)
                    blocks.add(parent)
                    break
            if parent_in_blocks is None:
                break
        return blocks






    data_node = get_tree_data_node(G, start)
    data_node = computeCNR(G, data_node)
    top_cnr_nodes = sorted([node[1]['data'] for node in G.nodes(data=True)],key=lambda x: x.cnr, reverse=True)
    avg_cnr = sum([node.cnr for node in top_cnr_nodes])/len(top_cnr_nodes)
    top_cnr_nodes = [node for node in top_cnr_nodes if node.cnr >= avg_cnr]
    transitive_closure = nx.transitive_closure(G,reflexive=False).edges()
    blocks = identify_main_content_blocks(G,top_cnr_nodes,transitive_closure)
    if 'body_0' in blocks:
        blocks.remove('body_0')
    allowed_categories = [ElementCategory.CONTENT_SECTIONING]
    blocks_data= [get_tree_data_node(G,block) for block in blocks]
    blocks_data = [block for block in blocks_data if block.category[0] in allowed_categories or block.parent.category[0] in allowed_categories]
    if len(blocks_data) == 0:
        return None
    main_content = max(blocks_data,key=lambda x: x.cnr_node.text_length)
    main_content.category = (main_content.category[0],"main") # TODO MORE logic of selecting final one
    return main_content.label
