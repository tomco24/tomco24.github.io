class HTMLSematicTreeGenerator():

    def __init__(self) -> None:
        super().__init__()
        self.generator = SemanticTreeGenerator()
        self.add_nodes_to_graph = False
        self.leafs = []

    def build_page_tree(self, soup, fname="output.svg",add_nodes_to_graph=False):
        #soup = BeautifulSoup(data, 'html.parser')
        tree,root_node = self.generator.build_page_tree(soup,start="body")
        self.add_nodes_to_graph = add_nodes_to_graph
        self.update_node_params(tree, root_node,None)
        tree.nodes[root_node]["leafs"] = self.leafs
        return tree

    def update_node_params(self,tree: nx.DiGraph,node,parent):
        data_node:SemanticNode = get_tree_data_node(tree, node)
        data_node.entity.values = []
        children = list(tree.successors(node))
        if parent:
            data_node.parent = get_tree_data_node(tree,parent)
        data_node.link_level = self.get_node_link_level(data_node)
        if len(children) == 0:
            res_extraction = self._extract_tags_texts(data_node)
            data_node.leaf = res_extraction
            if res_extraction:
                self.leafs = self.leafs + [data_node]
            if self.add_nodes_to_graph:
                for i,label in enumerate(data_node.labels_values):
                    label_data= data_node.entity.values[i]
                    #label_data = data_node.get_value_link(i)
                    tree.add_node(label, data=data_node.entity.values[i],label=label_data)
                    tree.add_edge(node, label)
        else:
            for child in children:
                self.update_node_params(tree, child,node)

    def _extract_tags_texts(self,node:SemanticNode):
        i = 0
        node.entity.values = []
        tags = []
        if node.label == 'h3_147.person-name_merged':
            print("here")
        for tag in node.tags:
            text = tag.get_text(strip=True)
            if len(text)>0:
                tags.append(tag)
                node.entity.values.append(text)
                node.labels_values.append(f"{node.label}_v{i}")
            i+=1
        if len(tags) == 0:
            return False
        node.tags = tags
        node.entity.setup_keyword_structure()
        node.entity.setup_ner_structure()
        return True
    def get_node_link_level(self,node:SemanticNode):
        if 'a' in node.path:
            index = node.path.index('a')
            level = len(node.path) - index
            return level
        return 0




class HTMLTreeGenerator():

    def __init__(self) -> None:
        super().__init__()
        self.counter = 0


    def parse(self, data: str, type: str):
        result = []
        soup = BeautifulSoup(data, 'html.parser')
        headings = soup.find_all(['h1', 'h2', 'h3', 'h4', 'h5', 'h6'])

        # loop through each tag in the soup

        for i, heading in enumerate(headings):
            heading_dict = {"heading": heading.name, "code": str(heading)}
            next_sibling = heading.find_next_sibling()
            while next_sibling and next_sibling.name not in ['h1', 'h2', 'h3', 'h4', 'h5', 'h6']:
                heading_dict["code"] += str(next_sibling)
                next_sibling = next_sibling.find_next_sibling()
            result.append(heading_dict)
        return result



    def build_page_tree(self, soup, fname="output.svg", prune_tree=True,):
        #soup = BeautifulSoup(data, 'html.parser')
        body = soup.find('body')
        G = nx.DiGraph()
        parent_node = 'body'
        data_node = EntityNode(self.counter, 'body')
        G.add_node(parent_node,data=data_node,label=parent_node)
        self._build_nx_node(body, parent_node, G)
        if prune_tree:
            pruned = self.prune_tree(G)
            if pruned:
                G = pruned
        A = nx.nx_agraph.to_agraph(G)
        A.layout('dot')
        A.draw(fname)
        return G

    def _build_nx_node(self, node, parent_node, G):
        self.counter += 1
        parent_data = G.nodes[parent_node]['data']
        for child in node.children:
            if child.name:
                #child_node = child.name + child.get("id",str(child.get("class","")))
                child_node = parent_node + " > " +child.name
                #child_node = child.get("unique_id")
                if not G.has_node(child_node):
                    data_node = EntityNode(self.counter, child.name, parent=parent_data)
                    data_node.add_tag(child)
                    G.add_node(child_node,data=data_node,label=child.name)
                else:
                    data_node = G.nodes[child_node]['data']
                    #G.nodes[child_node]['label'] =  child.get("role",'else') if G.nodes[child_node]['label']=='else' else G.nodes[child_node]['label']
                    data_node.add_tag(child)
                e =(parent_node, child_node)
                if not G.has_edge(*e):
                    G.add_edge(*e,label=1)
                else:
                    G[parent_node][child_node]['label'] +=1
                self._build_nx_node(child, child_node, G)
    def prune_tree(self,G:nx.DiGraph):
        start = "body"
        layers = nx.bfs_layers(G,start)
        node_main = None
        level = 1
        #layer = next(layers)
        main_layer = None
        for layer in layers:
            for node in layer :
                data_node = G.nodes.get(node).get("data")
                if data_node.is_main_content():
                    node_main = node
                    main_layer = list(layer)
                    return self.remove_non_main_branches(G,node_main,main_layer)
            level +=1


        #layer = list(nx.bfs_layers(G,start))[level]

        return None

    def remove_non_main_branches(self,G,node_main,layer):
        for node in layer:
            if node != node_main:
                self.remove_descendants(G,node)
    def remove_descendants(self,G, node):
        descendants = nx.descendants(G, node)
        descendants.add(node)
        G.remove_nodes_from(descendants)
