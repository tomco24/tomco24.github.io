class EntityNode:

    def __init__(self) -> None:
        super().__init__()
        self.values = None
        self.entities = {}
        self.keywords = {}
        self.ner_stats = defaultdict(int)
        self.keywords_stats = {}
        self.was_ner_tagged = False
        self.ner_tag = None
        self.keyword_tag = None
        self.child_ner_stats = defaultdict(int)

    def setup_ner_structure(self):
        self.entities = [{} for _ in self.values]

    def update_ner_tag(self):
        self.ner_tag = max(self.ner_stats, key=self.ner_stats.get)
        return self.ner_tag

    def setup_keyword_structure(self):
        self.keywords = [{} for _ in self.values]
        self.keywords_stats = {_: 0 for _ in keywords_map.keys()}

    def add_ner(self,results_ner):
        for i,result_ner in enumerate(results_ner):
            if result_ner:
                self.entities[i] = result_ner
                self.update_ner_status(result_ner)


    def update_ner_status(self, result_ner):
        for key in result_ner.keys():
            self.ner_stats[key] += len(result_ner[key])
        self.ner_tag = max(self.ner_stats, key=self.ner_stats.get)

    def num_tags_ner(self,type=None) -> int:
        if type:
            return sum([bool(item.get(type,False)) for item in self.entities])
        else:
            return sum([len(item.keys())>0 for item in self.entities])

    def update_parent_ner_stats(self):
        parent = self.parent
        child = self
        for key, value in child.ner_stats.items():
            parent.child_ner_stats[key] += value
        _ = parent.parent
        child = parent
        parent = _
        while parent:
            for key,value in child.child_ner_stats.items():
                parent.child_ner_stats[key] += value
            _ = parent.parent
            child = parent
            parent = _


    def add_keyword_match(self,idx,result_keyword):
        if idx > len(self.values):
            raise Exception("idx is out of bounds")
        self.keywords[idx] = result_keyword
        self.update_keywords_stats(result_keyword)

    def update_keywords_stats(self,result_keyword):
        for key in result_keyword.keys():
            self.keywords_stats[key] += len(result_keyword[key])
        self.keyword_tag = max(self.keywords_stats, key=self.keywords_stats.get)


    def features(self,id_):
        res = {
            "value":self.values[id_],
            "ner":self.entities[id_],
            "keywords":self.keywords[id_] if self.keywords else None

        }
        return res

    def search_in_values(self,search_string,threshold=2):
        result = []
        for i,value in enumerate(self.values):
            search_tokens = simple_word_tokenization(search_string)
            count = 0
            for token in search_tokens:
                if find_with_regex(value,token):
                    count+=1
            if count>threshold:
                result.append(i)
        return tuple(result)

class SemanticNode:

    def __init__(self, tag:bs4.Tag, parent: "SemanticNode") -> None:
        super().__init__()
        tag_id = tag.get("id","")
        tag_class = ".".join(tag.get("class", []))
        #self.type = f"{tag.name}#{tag_id}"
        self.type = tag.name
        self.id = None
        self.tag = tag
        self.tags = [tag]
        self.label = tag.name
        self.parent = parent
        self.category:tuple[ElementCategory,str] = (None, None)
        self.cnr_node = CNRNode(0,0)
        self.link_tag = True if tag.name == 'a' else False

        self.path = []
        self.path_str = None
        self.flatten = False
        self.replaced = []

        self.labels_values = []
        self.leaf = False
        self.link_level = 0

        self.entity = EntityNode()

    @property
    def cnr(self):
        return self.cnr_node.cnr


    def set_cnr(self,weight,length):
        self.cnr_node.weight = weight
        self.cnr_node.text_length = length
    def has_same_path(self, other):
        return self.path == other.path

    def merge(self, other, same=True):
        self.tags += other.tags
        self.link_tag = self.link_tag or other.link_tag
        # self.label +='.'+other.label
    def is_link_descendant(self):
        return 'a'==self.type or 'a' in self.path or self.link_tag
    def set_path_string(self):
        self.path_str = tuple([tuple([".".join(self.path)])])

    def __repr__(self) -> str:
        # return super().__repr__()
        return f"Node {self.label} {self.flatten} {self.replaced}"

    def get_serializable(self):
        res = {
            "path": str(self.path),
        }
        return res

    def get_tag_feature(self,id_):
        features = self.entity.features(id_)
        features.update({"tag":self.tags[id_].name})
        return features
    def get_tag_features(self):
        res = []
        for i in range(len(self.entity.values)):
            res.append(self.get_tag_feature(i))
        return res

    def get_ner_in_tag(self,ner_type,search_tag):
        result = []
        for i in range(len(self.entity.values)):
            features = self.get_tag_feature(i)
            if find_with_regex(features["tag"],search_tag):
                if ner_type in features["ner"]:
                    result.append(features)
        return result

    def get_value_link(self,idx):
        tag = self.tags[idx]
        if tag.name == 'a':
            return tag.get("href",None)
        parent_in_level = get_parent_by_level(tag, self.link_level)
        return parent_in_level.get("href",None)

    def get_full_tag_path(self, i):
        tag = self.tags[i]
        path = list(self.path)
        single_tag_id = f"#{tag.get('id','')}.{'.'.join(tag.get('class',[]))}"
        return tuple(path + [tag.name,single_tag_id])
