import numpy as np
import spacy
from numba import jit
import gensim.downloader as api
import gensim
from sklearn.metrics.pairwise import cosine_similarity
from academic_crawler.config.predefined import keywords_person_structure_positive, keywords_person_structure_negative


class WorldSimilarityWrapper:
    model: gensim.models.keyedvectors.KeyedVectors

    def __init__(self) -> None:
        super().__init__()
        # self.model = api.load("glove-twitter-25")
        # self.model.init_sims(replace=True)
        # self.model.save("./glove.model")
        self.model = gensim.models.keyedvectors.KeyedVectors.load(
            "/home/tomco24/Documents/Programming/dp/prototype/academic_crawler/academic_crawler/ml/glove.model",
            mmap="r")
        print("model_load")

    def most_similar(self, word):
        return self.model.similarity(word, "people")

    def similarity_between_word_lists(self, A, B):
        result = {}
        for itemA in A:
            if itemA not in self.model:
                result[itemA] = None
                continue
            row = []
            for itemB in B:
                if itemB not in self.model:
                    similarity = 0  # None? maybe TODO
                else:
                    similarity = self.model.similarity(itemA, itemB)
                row.append(tuple([itemB, similarity]))
            result[itemA] = row
        return result

    def get_max_similarity(self, records):
        if records is None:
            return 0
        else:
            maximum = max(records, key=lambda x: x[1])[1]
            if maximum < 0.9:
                return 0
            return maximum

    def get_similariy_with_person_structure_keywords(self, words):
        postiive = self.similarity_between_word_lists(words, keywords_person_structure_positive)
        negative = self.similarity_between_word_lists(words, keywords_person_structure_negative)
        return postiive, negative
