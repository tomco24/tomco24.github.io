from time import sleep

from academic_crawler.helpers.mock import get_mock_organization

import requests
from bs4 import BeautifulSoup,Tag
from academic_crawler.scrapers.templates.websites import configs


class ReferenceScraper:

    def __init__(self,page_config:PageConfig) -> None:
        super().__init__()
        self.page_config:PageConfig = page_config

    def save_people(self,people):
        with open("./FIT_reference_people.jsonl",'w') as f:
            for person in people:
                line = person.to_json() + "\n"
                f.write(line)
        return True

    def scrape(self):
        people = self.scrape_people()
        full_people = self.scrape_single_persons(people)
        self.save_people(full_people)






    def scrape_people(self):
        people = []
        url = self.page_config.people_url
        response = requests.get(url)
        sleep(1)
        organization = self.page_config.parent_organization
        soup = BeautifulSoup(response.text, 'html.parser')
        locations = self.page_config.people_tags_locations
        if locations.person_name[0] != locations.person_url[0]:
            raise Exception("Person name and url must be in the same tag")
        tag = locations.person_name[0]
        for container in soup.find_all(tag.name,tag.attrs):
            names = container.find_all(locations.person_name[1].name,locations.person_name[1].attrs)
            hrefs = container.find_all(locations.person_url[1].name,locations.person_url[1].attrs)
            if len(names)  != len(hrefs):
                raise Exception("Number of names and hrefs must be the same")
            for i in range(len(names)):
                hrefs[i] = hrefs[i].get('href')
                name = names[i].get_text(strip=True)
                url, type_ = organization.fix_url(hrefs[i])
                if type_ == "external":
                    continue
                person = Person(name=name,url=url.url,organization=organization)
                people.append(person)
        return people


    def get_name_from_people(self,tag:Tag,locations):
        i = 0
        name_tag =tag
        for location in locations:
            if location.closing:
                return name_tag.get_text(strip=True)
            name_tag = name_tag.find(location.name,location.attrs)
            if name_tag is None:
                return None

    def get_url_from_people(self, tag: Tag, locations):
        i = 0
        name_tag = tag
        for location in locations:
            if location.closing:
                return name_tag.get('href')
            name_tag = name_tag.find(location.name, location.attrs)
            if name_tag is None:
                return None

    def scrape_single_persons(self, people):
        single_person_locations = self.page_config.single_person_tags_locations
        container = single_person_locations.affiliation_container
        division_location = single_person_locations.affiliation_division
        position_location = single_person_locations.affiliation_position
        for person in people:
            response = requests.get(person.url)
            soup = BeautifulSoup(response.text, 'html.parser')
            try:
                tag_container = soup.find(container.name,container.attrs)
                if container.name == "table":
                    self.scrape_table(person,tag_container,division_location,position_location)
                print(person)
            except Exception as e:
                print(e)
                print(person)
        return people
    def scrape_table(self, person, tag_container, division_location, position_location):
        for row in tag_container.find_all("tr"):
            columns = row.find_all("td")
            if len(columns) < 2:
                continue
            division = columns[division_location.attrs.get('position')].get_text(strip=True)
            position = columns[position_location.attrs.get('position')].get_text(strip=True)
            person.add_affiliation(division,position)
