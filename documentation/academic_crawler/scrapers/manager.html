from furl import furl

from academic_crawler.config.page_types import PAGE_TYPE
from academic_crawler.heuristics.links import LinkManager
from academic_crawler.scrapers.templates import Organization
from enum import Enum

class ScrapingState(Enum):
    PARSE_HOME = 1
    SCRAPE_ORGS = 2
    SCRAPE_ORGS_PEOPLE = 3
    SCRAPE_PEOPLE = 4
    SCRAPE_PARTIAL_PEOPLE = 5
    SCRAPE_OTHER = 6
    SCRAPE_FINISHED = 7


class ScrapingManager:

    def __init__(self) -> None:
        super().__init__()
        self.patterns = {
            "org":None,
            "people_page":None,
            "person":None,
        }
        self.count = 0
        self.results = {
            "orgs":defaultdict(list),
            "people":defaultdict(list),
        }
        self.visited = {
            "orgs":[],
            "people":[],
        }
        self.states = {state: {"managers": [], "callback": None, "instances": [], "page_types":{},"urls":set(),"instance_urls":set()}
                         for state in ScrapingState if state.name.startswith("SCRAPE")}
        self.actual = ScrapingState.PARSE_HOME
        self.partial_people = []
        self.partial_length = 0

    def setup_states(self, orgs_managers,people_managers,other_managers):
        self.states[ScrapingState.SCRAPE_ORGS]["managers"] = orgs_managers
        self.states[ScrapingState.SCRAPE_ORGS]["page_types"] = {PAGE_TYPE.PAGE_ORGANIZATIONS}
        self.states[ScrapingState.SCRAPE_ORGS_PEOPLE]["page_types"] = { PAGE_TYPE.PAGE_PEOPLE_NO_DATA, PAGE_TYPE.PAGE_PEOPLE_DATA}
        self.states[ScrapingState.SCRAPE_PEOPLE]["managers"] = people_managers
        self.states[ScrapingState.SCRAPE_PEOPLE]["page_types"] = {PAGE_TYPE.PAGE_PEOPLE_NO_DATA, PAGE_TYPE.PAGE_PEOPLE_DATA}
        self.states[ScrapingState.SCRAPE_OTHER]["managers"] = other_managers
        self.states[ScrapingState.SCRAPE_OTHER]["page_types"] = {PAGE_TYPE.PAGE_PEOPLE_NO_DATA, PAGE_TYPE.PAGE_PEOPLE_DATA}



    def add_org(self,org:Organization,idx):
        self.results["orgs"][idx].append(org)
    def add_person(self,person,idx):
        self.results["people"][idx].append(person)

    def add_partial_idx(self,idx,length):
        self.partial_people.append(idx)
        self.partial_length += length
    def add_visited_org_page(self, manager:LinkManager):
        self.visited["orgs"].append(manager)
        return len(self.visited["orgs"]) -1
    def add_visited_people(self, manager:LinkManager):
        self.visited["people"].append(manager)
        return len(self.visited["people"]) -1

    def get_all_partial_persons(self):
        return [(idx,self.results["people"][idx]) for idx in self.partial_people]
    def get_orgs_urls(self,orgs):
        urls = []
        for org in orgs:
            urls.append(org.url)
        return urls

    def get_people_urls(self,people):
        urls = []
        for person in people:
            urls.append(person.url)
        return urls


    def extract_url_pattern(self,urls):
        matches = []
        for i in range(len(urls)):
            for j in range(i+1,len(urls)):
                url_a = urls[i]
                url_b = urls[j]
                seqMatch = SequenceMatcher(None, url_a, url_b)
                match = seqMatch.find_longest_match(0, len(url_a), 0, len(url_b))
                if match.size > 0:
                    matches.append(url_a[match.a: match.a + match.size])
        counter = Counter(matches)
        most_common = counter.most_common(1)
        pattern = f"{most_common[0][0]}*"
        return pattern

    def build_structure_and_patterns(self):
        orgs = self.get_instances(ScrapingState.SCRAPE_ORGS)
        orgs_people = self.get_instances(ScrapingState.SCRAPE_ORGS_PEOPLE)
        people = self.get_instances(ScrapingState.SCRAPE_PEOPLE)
        other_people = self.get_instances(ScrapingState.SCRAPE_OTHER)
        self.count = len(orgs_people) + len(people) + len(other_people)
        return people,orgs_people,other_people
        #return orgs,orgs_people,people


    def get_managers(self, state):
        return self.states[state]["managers"]

    def get_callback(self, state):
        return self.states[state]["callback"]

    def get_page_types(self, state):
        return self.states[state]["page_types"]

    def add_url_to_state(self, state, url):
        self.states[state]["urls"].add(url)

    def add_instance_to_state(self,state,instance):
        if isinstance(instance,Organization) and instance.url in self.states[state]["instance_urls"]:
            return
        self.states[state]["instance_urls"].add(instance.url)
        self.states[state]["instances"].append(instance)

    def add_manager_to_state(self,state,manager):
        if isinstance(manager,tuple) and len(manager[0]) == 0:
            return
        self.states[state]["managers"].append(manager)

    def get_instances(self, state):
        return self.states[state]["instances"]
