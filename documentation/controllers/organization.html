import enum
from academic_crawler.config.domain import DomainConfiguration,SubsidiaryConfiguration
from transitions import Machine

class StatesOrganization(enum.Enum):
    START = 0
    PARSE_HOME_LINKS = 1
    DETECT_KEYWORDS = 2
    PARSE_CANDIDATES = 3
    PARSING_ALL = 4
    FAILED = 5
    RESOLVE_CANDIDATES = 6
    BUILD_ORG_STRUCTURE = 7


class OrganizationController:

    def __init__(self, state=None) -> None:
        super().__init__()
        if state is None: # TODO REMOVE THIS
            config = dict()
        self.state = state
        self.spider = None
        self.approval = False

    def has_entity_match(self):
        return self.state.get("has_entity_match")

    def has_approval(self):
        return self.approval


def build_state_machine(config={}):
    controller = OrganizationController(config)
    machine = Machine(controller, StatesOrganization,transitions=[], initial=StatesOrganization.START) #after_state_change="state_change"
    machine.add_transition("move", StatesOrganization.START, StatesOrganization.PARSE_HOME_LINKS)
    machine.add_transition("move", StatesOrganization.PARSE_HOME_LINKS, StatesOrganization.PARSE_CANDIDATES,conditions=['has_approval'])
    machine.add_transition("move", StatesOrganization.PARSE_HOME_LINKS, StatesOrganization.PARSING_ALL,unless=['has_approval'])
    machine.add_transition("move",StatesOrganization.PARSE_CANDIDATES,StatesOrganization.RESOLVE_CANDIDATES)
    machine.add_transition("move",StatesOrganization.RESOLVE_CANDIDATES,StatesOrganization.BUILD_ORG_STRUCTURE)
    #machine.add_transition("move", StatesOrganization.PARSE_HOME_LINKS, StatesOrganization.DETECT_KEYWORDS)
    machine.add_transition("move", StatesOrganization.DETECT_KEYWORDS, StatesOrganization.PARSE_CANDIDATES, conditions=["has_entitity_match"])
    machine.add_transition("move",StatesOrganization.DETECT_KEYWORDS,StatesOrganization.FAILED,unless=["has_entitity_match"]) # NO MATCHED KEYWORDS OR ENTITIES
    machine.add_transition("move",StatesOrganization.FAILED,StatesOrganization.PARSING_ALL,conditions=[])


    return controller

    #machine.add_transition("move",States.FILTER_LINKS,States.SCRAPE_PERSON_PAGE)



    return machine,controller
