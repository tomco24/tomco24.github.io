class Organization:
    name: str
    domain:str
    url: str
    parent : Optional["Organization"]
    abbreviation: str = ""
    #children: list["Organization"]

    def fix_url(self,url):
        base = furl(self.url)
        url = furl(url)
        if not url.host:
            res =base.join(url)
            return res,"relative"
        elif self.is_subdomain(url.url):
            return url,"subdomain"
        elif base.host == url.host:
            return url,"absolute"
        else:
            return url,"external"




    def is_subdomain_2(self, url):
        base = furl(self.url)
        url = furl(url)
        if url.host == base.host:
            return False
        url_subdomains = url.host.split('.')[1:]
        base_url_subdomains = base.host.split('.')[1:]
        if len(url_subdomains) - len(base_url_subdomains) == 1:
            same = all([base_url_subdomains[i] == url_subdomains[i+1] for i in range(len(base_url_subdomains))])
            return same
        return False

    def is_subdomain(self,url):
        domain = tldextract.extract(self.domain)
        domain_name = domain.domain
        url_domain = tldextract.extract(url)
        subdomains = url_domain.subdomain.split('.')
        if 'www' in subdomains:
            subdomains.remove('www')
        common = set(subdomains).intersection(set(domain.subdomain.split('.')))
        subdomains = [sub for sub in subdomains if sub not in common]
        if url_domain.domain == domain_name and len(subdomains) > 0:
            return True
        return False
