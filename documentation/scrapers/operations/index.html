from collections import namedtuple
from academic_crawler.config.enums import OperationStatus, OperationType
from dataclasses import dataclass
import logging
from functools import wraps


@dataclass
class OperationMessage:
    type: OperationType
    status: OperationStatus
    has_information_data: bool


@dataclass
class OperationResult:
    data: dict
    message: OperationMessage

#OperationResult = namedtuple("OperationResult", ['data', 'message'])
#OperationMessage = namedtuple("OperationMessage", ['type', 'status', 'has_information_data'])


def op_success(result: OperationResult):
    return result.message.status == OperationStatus.SUCCESS


def op_failed(result: OperationResult):
    return result.message.status == OperationStatus.FAILED

def get_op_status(expression,is_list=False):
    if is_list and not expression:
        return OperationStatus.EMPTY_RESULT
    return OperationStatus.SUCCESS if expression else OperationStatus.FAILED


def make_operation_message(type=None, status=None, has_information_data=False):
    return OperationMessage(type=type, status=status, has_information_data=has_information_data)

def create_operation_result(func) ->OperationResult:
    @wraps(func)
    def wrap(*args, **kwargs):
        f_result = func(*args, **kwargs)
        if len(f_result) == 2:
            data, type_ = f_result
            if isinstance(data,type_.output_type):
                status = get_op_status(True)
                if type_.output_type is list:
                    status = get_op_status(len(data)>0,is_list=True)
            else:
                status = get_op_status(False)
            has_information = data is not None
            message = OperationMessage(type_, status, has_information)
            logging.info(f"Operation {type_} finished with status {status} and data {data}")
            return OperationResult(data, message)
        if len(f_result) == 3:
            data, type_, status = func(*args, **kwargs)
            has_information = data is not None
            message = OperationMessage(type_, status, has_information)
            logging.info(f"Operation {type_} finished with status {status} and data {data}")
            return OperationResult(data, message)
    return wrap
