import logging
from pprint import pprint

from . import BaseScrapingStrategy
from ..operations.people import confirm_name_on_single_page, extract_affiliations_candidates_from_single_page, \
    filter_affiliation_candidates, extract_affiliation_elements, get_siblings_for_tag
from ...config.page_types import PAGE_TYPE
from ...config.utils import get_relabeled_tree
from ..operations import op_success
from ...heuristics.tree import SemanticNode


class SinglePersonStrategy(BaseScrapingStrategy):


    def __init__(self,input_data,patterns) -> None:
        super().__init__()
        self.patterns = patterns
        self.page_header = input_data.get("state").page_header
        self.page_type: PAGE_TYPE = input_data.get("page_type")
        # self.parent_org: input_data.get("state").get_domain("organization")
        self.tree = self.page_header.get("tree")
        self.organization =input_data.get("state").get_domain("organization")
        self.person = input_data.get("person")
        self.vocab_manager = input_data.get("state").get("vocab_manager")
    def run(self):
        #TODO ADD EMAIL
        relabeled_tree = get_relabeled_tree(self.tree)
        op_res = confirm_name_on_single_page(relabeled_tree,self.person)
        if not op_success(op_res):
            logging.error("Could not confirm name on single page")
        name_nodes = op_res.data
        leafs: list[SemanticNode] = relabeled_tree.nodes[0]['leafs']
        for leaf in leafs:
            op_res = extract_affiliations_candidates_from_single_page(leaf, self.person, name_nodes)
            if not op_success(op_res):
                logging.error("Could not extract affiliations candidates from tree leaf")
            else:
                candidates = op_res.data
                op_res = filter_affiliation_candidates(relabeled_tree, self.person, candidates,self.vocab_manager)
                if not op_success(op_res):
                    logging.error("Could not filter affiliations candidates from tree leaf")
                    continue
                affiliations = op_res.data
                for affiliation in affiliations:
                    self.person.add_affiliation(affiliation[0],affiliation[1])
        return self.person



    def run_alternative(self):
        relabeled_tree = get_relabeled_tree(self.tree)
        leafs: list[SemanticNode] = relabeled_tree.nodes[0]['leafs']
        for leaf in leafs:
            op_res = extract_affiliation_elements(leaf, self.person,vocab_manager=self.vocab_manager)
            if op_success(op_res):
                affiliations,patterns = op_res.data
                self.patterns += patterns
                for affiliation in affiliations:
                    self.person.add_affiliation(affiliation.division, affiliation.position)
        return self.person

    def run_patterns(self):
        relabeled_tree = get_relabeled_tree(self.tree)
        leafs: list[SemanticNode] = relabeled_tree.nodes[0]['leafs']
        for leaf in leafs:
            self.extract_affiliations_from_patterns(leaf,self.person)
        return self.person


    def _match_sibling(self,filter_siblings,pattern_path):
        for siblings in filter_siblings:
            for sibling in siblings:
                path = sibling.get("doc_path")
                if path == pattern_path:
                    return sibling
    def extract_affiliations_from_patterns(self,leaf,person):
        limit = 1
        for tag in leaf.tags:
            doc_path = tag.get("doc_path")
            for pattern in self.patterns:
                pattern_type,pattern_paths = pattern
                is_value = len(pattern_paths)>1 and doc_path == pattern_paths[1]
                if is_value and pattern_type == "key_value":
                    pre_siblings, post_sibling = get_siblings_for_tag(tag)
                    filter_siblings = [pre_siblings[:limit], post_sibling[:limit]]
                    matched_sibling = self._match_sibling(filter_siblings,pattern_paths[0])
                    if matched_sibling:
                        person.add_affiliation(tag.get_text(strip=True),matched_sibling.get_text(strip=True))
                        break



    def extract_template_instances(self) -> list:
        pass
