class OrganizationSpider(AcademicSpider):
    name = 'OrganizationSpider'
    custom_settings = {
        'ITEM_PIPELINES': {
        },
    }

    def __init__(self, *a, **kw):
        super().__init__(*a, **kw)
        self.link_extractor = LinkExtractor(allow_domains=self.allowed_domains, unique=True)#, allow=url_patterns)

        self.scraper = OrgStructureScraper()
        self.setup_state_callback_map()

        self.next_state()

    def setup_state_callback_map(self):
        self.state_callback_map = {
            StatesOrganization.PARSE_HOME_LINKS:self.parse_home,
            StatesOrganization.PARSE_CANDIDATES:self.parse_candidate,

        }



    def start_requests(self):
        if len(self.start_urls) == 0:
            return []
        url = self.start_urls[0]
        yield SplashRequest(url, self.get_state_callback(), args={'wait': 1.5})

    def parse_home(self, response, **kwargs):
        logging.info(f"Starting to scrape home page of domain{ self.allowed_domains[0]}")
        managers = self.get_page_links(response,{})
        candidates = []
        for manager in managers:
            if manager.keyword_entity_match(self.detectors.get('keyword_category'),"keywords_org_structure"):
                if len(manager.get_referer_text_tokens()) <=5:
                    candidates.append(manager)

        if len(candidates)>0:
            self.next_state()
            i = len(candidates) +1
            for manager in candidates:
                i -=1
                yield SplashRequest(manager.link.url,self.get_state_callback(),args={'wait': 1.5},priority=i)
        else:
            self.next_state(False)

    def parse_candidate(self,response,**kwargs):
        """
        This method tries to extract Organization instances from the given response page
        Args:
            response:
            **kwargs:

        Returns:

        """
        self.generate_page_header(response, prune_tree=False)
        pipeline = PageFeaturesPipeline(self.program_state)
        page_type = pipeline.execute()
        if page_type != PAGE_TYPE.PAGE_ORGANIZATIONS:
            return
        #save_graph(self.program_state.get_page("tree"),self.program_state.get_page("title")+".gexf")
        candidate = {
            "page_type":page_type,
            "state":self.program_state,
        }
        instances =self.scraper.scrape(candidate)
        if len(instances)>0:
            internal_instances = instances.get("internal")
            for instance in internal_instances:
                yield SplashRequest(instance.url, self.look_for_subdomain_link, args={"wait":1.5}, meta={"instance":instance})
            children_instances = instances.get("children")
            for instance in children_instances:
                yield {'instance':instance, 'type':'ORGANIZATION', 'status':'FULL'}

    def look_for_subdomain_link(self,response):
        """
        This method tries to find a link to the subdomain of the Organization instance
        Args:
            response:

        Returns:

        """
        instance = response.meta.get("instance")
        managers = self.get_page_links(response,{})
        parent_org = self.program_state.get_domain("organization")
        search_query = f"{parent_org.name} - {instance.name}"
        web_search_domains = get_search_query_domains(search_query)
        most_likely_domain = web_search_domains.most_common(1)[0][0]
        logging.warning(f"Most likely domain for {search_query} is {most_likely_domain} with {web_search_domains.most_common(1)[0][1]} results")
        candidates = []
        for manager in managers:
            if parent_org.is_subdomain(manager.link.url) and most_likely_domain in manager.link.url:
                logging.info(
                    f"Checking link {manager.link.url} for subdomain of {parent_org.name} ({parent_org.domain})")
                candidates.append(manager)
        candidates_filtered = [candidate for candidate in candidates if 'en' in candidate.link.url]  # TODO BETTER FIX
        if len(candidates_filtered) > 0:
            candidates_sorted = sorted(candidates_filtered, key=lambda x: len(x.link.url))
            manager = candidates_sorted[0]
            instance.url = manager.link.url
            instance.domain = manager.url.host
            self.scraper.add_instance(instance)
            yield {'instance': instance, 'type': 'ORGANIZATION', 'status': 'FULL'}
        else:
            yield {'instance': instance, 'type': 'ORGANIZATION', 'status': 'FULL'}
