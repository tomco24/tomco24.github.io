class PAGE_TYPE(Enum):
    PAGE_PEOPLE_DATA = 1
    PAGE_PEOPLE_NO_DATA = 2
    PAGE_SINGLE_PERSON = 3
    PAGE_ORGANIZATIONS = 4
    PAGE_GENERAL = 5



class PageTypeMatcher:


    page_dict: dict

    DEFAULT_DICT = page_type_default_dict
    NER_THRESHOLD_NAME = 5
    NER_THRESHOLD_DEPARTMENT = 3

    def __init__(self,to_exclude=[]) -> None:
        super().__init__()
        self.page_dict = dict(PageTypeMatcher.DEFAULT_DICT)
        self.last_score = 0
        self.to_exclude = to_exclude

        self.separations = {
            PAGE_TYPE.PAGE_PEOPLE_DATA:((0,3),(3,None)),
            PAGE_TYPE.PAGE_PEOPLE_NO_DATA:((0, 3),(None,None)),
            PAGE_TYPE.PAGE_SINGLE_PERSON:((0,3),(3,None)),
            PAGE_TYPE.PAGE_ORGANIZATIONS:((0, 4), (4, None)),
        }
    def update_page_dict(self,new_dict):
        self.page_dict.update(new_dict)
        self.condition_dictionary = {
            PAGE_TYPE.PAGE_PEOPLE_DATA: (
                self.page_dict.get("num_names_page") > PageTypeMatcher.NER_THRESHOLD_NAME,
                not self.page_dict.get("has_names_in_links"),
                not self.page_dict.get("has_siblings_in_links_name"),
                self.page_dict.get("has_email"),
                self.page_dict.get("has_keyword_position_match"),
                self.page_dict.get("has_pagination")
            ),
            PAGE_TYPE.PAGE_PEOPLE_NO_DATA: (
                self.page_dict.get("num_names_page") >= PageTypeMatcher.NER_THRESHOLD_NAME,
                self.page_dict.get("has_names_in_links"),
                self.page_dict.get("num_org_nodes") <= self.page_dict.get("num_name_nodes"),

            ),
            PAGE_TYPE.PAGE_SINGLE_PERSON: (
                self.page_dict.get("num_names_page") <= PageTypeMatcher.NER_THRESHOLD_NAME - 1,
                self.page_dict.get("has_name_in_heading"),
                self.page_dict.get("has_name_in_title"),
                self.page_dict.get("has_email"),
                self.page_dict.get("has_keyword_position_match"),
            ),
            PAGE_TYPE.PAGE_ORGANIZATIONS: (
                self.page_dict.get("num_orgs_page") > PageTypeMatcher.NER_THRESHOLD_DEPARTMENT,
                self.page_dict.get("num_org_nodes") > self.page_dict.get("num_name_nodes"),
                self.page_dict.get("has_siblings_in_links_org"),
                self.page_dict.get("has_keyword_org_match"),
                self.page_dict.get("pagination"),
            ),

        }
        self.exclude_page_types(self.to_exclude)

    def exclude_page_types(self,to_exclude):
        for key in to_exclude:
            del self.condition_dictionary[key]
            del self.separations[key]
    def get_condition_score(self,conditions,weights=None):
        result = 0
        i = 0
        for condition in conditions:
            if condition:
                result += 1
            i +=1
        result = result/len(conditions)
        self.last_score = result
        return result

    def check_page_type(self,page_type:PAGE_TYPE,stage):
        conditions = self.condition_dictionary.get(page_type)
        separation = self.separations.get(page_type)
        main,other = [conditions[start:end] for start,end in separation]
        if stage == 1:
            return self.get_condition_score(main)
        elif stage ==2:
            return self.get_condition_score(other)

    def solve_page_type(self):
        print(self.page_dict)
        stage = 1
        scores = []
        if not self.has_enough_ner() or (not self.is_people_site() and not self.is_org_site()):
            return PAGE_TYPE.PAGE_GENERAL
        for page_type_ in self.separations.keys():
            scores.append((page_type_,self.check_page_type(page_type_,stage)))
        result = [item for item in scores if item[1]==1]
        if len(result) == 1:
            return result[0][0]
        elif len(result)>1:
            stage = 2
            other_scores = [(item[0],item[1] + self.check_page_type(item[0],stage)) for item in result]
            return max(other_scores,key=lambda x:x[1])[0]
        elif len(result) ==0:
            return PAGE_TYPE.PAGE_GENERAL
            stage = 2
            other_scores = [(item[0],item[1] + self.check_page_type(item[0],stage)) for item in scores]
            max_item = max(other_scores,key=lambda x:x[1])
            if max_item[1] > 1:
                return max_item[0]
            else:
                return PAGE_TYPE.PAGE_GENERAL









    def match_page_type_old(self):
        if self.check_page_people_data():
            return PAGE_TYPE.PAGE_PEOPLE_DATA
        elif self.check_page_people_no_data():
            return PAGE_TYPE.PAGE_PEOPLE_NO_DATA
        elif self.check_page_single_person():
            return PAGE_TYPE.PAGE_SINGLE_PERSON
        elif self.check_page_departments():
            return PAGE_TYPE.PAGE_ORGANIZATIONS
        elif self.check_general_page():
            return PAGE_TYPE.PAGE_GENERAL
        else:
            return None
    def match_page_type(self):
        result =[]

        if self.is_org_site():
            if self.check_page_departments():
                return PAGE_TYPE.PAGE_ORGANIZATIONS
        elif self.is_people_site():
            if self.check_page_people_data():
                result.append((PAGE_TYPE.PAGE_PEOPLE_DATA,self.last_score))
            if self.check_page_people_no_data():
                result.append((PAGE_TYPE.PAGE_PEOPLE_NO_DATA,self.last_score))
        elif self.check_page_single_person():
            result.append((PAGE_TYPE.PAGE_SINGLE_PERSON,self.last_score))
        else:
            if self.check_general_page():
                result.append((PAGE_TYPE.PAGE_GENERAL, self.last_score))

        if len(result)>0:
            return max(result,key=lambda x:x[1])[0]
        return None

        conditions =(

        main = (0,2)
        additional = (3,5)
        score = self.get_condition_score(conditions,(10,10,10,8,5,2))
        return score>.67

        conditions = (self.page_dict.get("num_names_page") >= PageTypeMatcher.NER_THRESHOLD_NAME,

        score = self.get_condition_score(conditions,(10,10))
        return score>0

    def has_enough_ner(self):
        if self.page_dict.get("num_names_page") >= PageTypeMatcher.NER_THRESHOLD_NAME or self.page_dict.get("num_orgs_page") >= PageTypeMatcher.NER_THRESHOLD_DEPARTMENT:
            return True
        return False
    def is_people_site(self):
        if self.page_dict.get("num_orgs_page") == 0:
            ratio = self.page_dict.get("num_name_nodes")
        else:
            ratio = self.page_dict.get("num_name_nodes")/max(self.page_dict.get("num_org_nodes",1),1)
        return ratio>=1 and  self.page_dict.get("num_names_page") > PageTypeMatcher.NER_THRESHOLD_NAME

    def is_org_site(self):
        if self.page_dict.get("num_names_page") == 0:
            ratio = self.page_dict.get("num_org_nodes")
        else:
            ratio = self.page_dict.get("num_org_nodes")/max(self.page_dict.get("num_name_nodes",1),1)
        return ratio >=1 and self.page_dict.get("num_orgs_page") > PageTypeMatcher.NER_THRESHOLD_DEPARTMENT

def match_page_type(self):
    result =[]

    if self.is_org_site():
        if self.check_page_departments():
            return PAGE_TYPE.PAGE_ORGANIZATIONS
    elif self.is_people_site():
        if self.check_page_people_data():
            result.append((PAGE_TYPE.PAGE_PEOPLE_DATA,self.last_score))
        if self.check_page_people_no_data():
            result.append((PAGE_TYPE.PAGE_PEOPLE_NO_DATA,self.last_score))
    elif self.check_page_single_person():
        result.append((PAGE_TYPE.PAGE_SINGLE_PERSON,self.last_score))
    else:
        if self.check_general_page():
            result.append((PAGE_TYPE.PAGE_GENERAL, self.last_score))

    if len(result)>0:
        return max(result,key=lambda x:x[1])[0]
    return None
