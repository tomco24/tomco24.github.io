import re
import numpy as np
import networkx as nx
from anytree import NodeMixin, RenderTree
from academic_crawler.config.utils import simple_word_tokenization

from academic_crawler.heuristics.links import LinkManager
from academic_crawler.ml.words import WorldSimilarityWrapper


class SiteNode(NodeMixin):

    def __init__(self, name, url, text, parent=None, children=None) -> None:
        super().__init__()
        self.name = name
        self.url = url
        self.text = text
        self.confidence = -1
        self.scores = []

    def format_node_name(self):
        self.name = f"{self.name}\n_______\n{self.text}"

    def get_score(self):
        if len(self.scores) == 0:
            return -1
        weights = [x + 1 for x in range(len(self.scores))]
        result = np.average(self.scores, weights=weights)
        return result


class SiteTreeManager:
    def __init__(self) -> None:
        super().__init__()
        self.tree = nx.DiGraph()
        self.sites = []
        self.root = None
        self.levels = []
        self.evaluator = TreeEvaluator()

    def _parse_path_to_level(self, path, level):
        if len(path) > level:
            return path
        return path[:(level + 1)]

    def _make_site_node(self, manager: LinkManager, level):
        result = SiteNode(manager.path[level], manager.link.url, manager.referer_text)
        return result

    def export_tree(self, name='tree.png'):
        import matplotlib.pyplot as plt
        nx.draw(self.tree, with_labels=True)
        plt.savefig("tree.png")
        nx.write_gpickle(self.tree, "tree2.pkl")

    def add_root(self, url):
        node = SiteNode("ROOT", url, "ROOT")
        self.tree.add_node("ROOT", data=node)
        self.root = node

    def build_tree(self, managers, max_level=3):
        if self.root is None:
            raise Exception("Root Node not set!")
        paths = [self._parse_path_to_level(x.path, max_level) for x in managers]  # TODO NOT INLINE
        levels = [{} for x in range(max_level)]
        for i in range(max_level):
            for j, path in enumerate(paths):
                if len(path) < (i + 1):
                    continue
                title_level = path[i]
                if len(title_level) == 0:
                    continue
                if i == 0:
                    if title_level not in levels[i].keys():
                        node = self._make_site_node(managers[j], i)
                        parent = self.root
                        # node.format_node_name()
                        levels[i][title_level] = node
                        self.tree.add_node(node.name, data=node)
                        self.tree.add_edge(parent.name, node.name)

                else:
                    if title_level not in levels[i].keys():
                        node = self._make_site_node(managers[j], i)
                        parent = levels[i - 1].get(path[i - 1], None)
                        # node.format_node_name()
                        levels[i][title_level] = node
                        self.tree.add_node(node.name, data=node)
                        self.tree.add_edge(parent.name, node.name)
        self.levels = levels
        return self.root

    def get_links_to_scrape(self):
        self.tree = self.evaluator.evaluate_nodes(self.tree)
        links = self.evaluator.select_paths(self.tree)
        return links


class TreeEvaluator:

    def __init__(self) -> None:
        super().__init__()
        self.sim_manager = WorldSimilarityWrapper()

    def get_final_score(self, values):
        positive = values.get('positive')
        negative = -1 * values.get('negative')
        return positive + negative

    def evaluate_nodes(self, G: nx.Graph, start_node="ROOT"):
        succesors = nx.dfs_successors(G, start_node)
        values = {start_node: 0}
        for parent, children in succesors.items():
            parent_node = G.nodes[parent]['data']
            for child in children:
                child_node = G.nodes[child]['data']
                child_node.scores = list(parent_node.scores)
                match_keys = ['positive', 'negative']
                child_values = {}
                sentence = re.sub('[-_]+', ' ', child)
                tokens = simple_word_tokenization(sentence, remove_stop=True)
                similarites = self.sim_manager.get_similariy_with_person_structure_keywords(tokens)
                for i in range(len(similarites)):
                    match_value_sum = 0
                    for token in tokens:
                        records = similarites[i].get(token)
                        match_val = self.sim_manager.get_max_similarity(records)
                        match_value_sum += match_val
                    match_value = match_value_sum / len(tokens) if len(tokens) > 0 else 0
                    child_values[match_keys[i]] = match_value
                final_score = self.get_final_score(child_values)
                child_node.scores.append(final_score)
                # print(f"Node:{child_node.name}:/t {child_node.get_score()}")

        return G

    def select_paths(self, G, start_node="ROOT"):
        layers = nx.bfs_layers(G, start_node)
        print(len(list(layers)))
        depth = 0
        children = G.successors(start_node)
        result = []
        for vertex in children:
            node = G.nodes[vertex]['data']
            score = node.get_score()
            print(vertex, score, depth)
            if score >= 0:
                res = self.explore_node(G, vertex, node, depth + 1)
                if res:
                    result = result + res
        result = sorted(result, key=lambda x: x.get_score(), reverse=True)
        result = [node.url for node in result]
        return result

    def explore_node(self, G, name, node, depth):
        if depth >= 3 and node.get_score() > 0:
            return [node]
        children = list(G.successors(name))
        result = []
        if len(children) == 0 and node.get_score() > 0:
            return [node]
        for vertex in children:
            node = G.nodes[vertex]['data']
            score = node.get_score()
            print(vertex, score, depth)
            if score >= 0:
                res = self.explore_node(G, vertex, node, depth + 1)
                if len(res) > 0:
                    result = result + res
        return result
