class AcademicAbbreviationDetector():
    def __init__(self) -> None:
        super().__init__()
        self.degrees = ['Bc.', 'Mgr.', 'Ing.', "Ph.D.", 'CSc.', 'RNDR.', 'MUDR.', "PhDr."]
        self.degrees_lower = [x.lower() for x in self.degrees]

    def is_academic_degree(self, text: str):
        r = re.compile(text, re.I)
        for degree in self.degrees:
            res = r.search(degree)
            if res:
                return True
        return False

    def is_degree_in_text(self, text):
        for degree in self.degrees:
            r = re.compile(degree, re.I)
            res = r.search(text)
            if res:
                return True
        return False


def is_email(text):
    r = re.compile(r"^\S+@\S+\.\S+$")
    if r.search(text):
        return True
    return False

class NERDetector:
    def __init__(self,types:set) -> None:
        super().__init__()
        self.types = types
        self.nlp = NlpSingleton.get_instance()
        self.has_email ='EMAIL' in types
        self.search_email = self._search_email if self.has_email else lambda x, y, z:None




    def extract_ner_single(self,text):
        entities = [defaultdict(list)]
        doc = self.nlp(text)
        for entity in doc.ents:
            if entity.label_ in self.types:
                entities[0][entity.label_].append(entity)
        entities,total_email = self.search_email(doc, 0,entities)
        return entities[0]
    def extract_ner(self, texts, intervals=None):
        entities =[ defaultdict(list) for _ in range(len(texts))]
        max_length = max([len(item) for item in texts])
        self.nlp.max_length = max_length + 100
        i = 0
        total_found = 0
        for doc in self.nlp.pipe(texts,disable=["tok2vec", "tagger", "parser", "attribute_ruler", "lemmatizer"]):
            for entity in doc.ents:
                if entity.label_ in self.types:
                    total_found +=1
                    entities[i][entity.label_].append(entity)
            entities,total_email = self.search_email(doc,i,entities)
            total_found += total_email
            i+=1


        self.nlp.max_length = 1000000
        return entities if total_found>0  else None

    def _search_email(self, doc, idx, entities):
        total_found_email = 0
        for token in doc:
            if token.like_email:
                total_found_email += 1
                entities[idx]['EMAIL'].append(token)
                print(token)
        return entities,total_found_email







class KeywordDetector():
    def __init__(self,vocabulary_manager=None) -> None:
        super().__init__()
        self.nlp = NlpSingleton.get_instance()
        self.ss = SnowballStemmer(language='english')
        self.manager = vocabulary_manager


    def extract_keyword(self,text):
        result =defaultdict(list)

        for key, keywords in keywords_map.items():
            for keyword in keywords:
                if re.search(r"\b" + keyword + r"\b", text, re.I):
                    result[key].append( keyword)
                else:
                    keyword_stem = self.ss.stem(keyword)
                    tokens = simple_word_tokenization(text)
                    for token in tokens:
                        if self.ss.stem(token) == keyword_stem:
                            result[key].append(keyword)
        return result

    def is_keyword(self,text):

        res = self.extract_keyword(text)
        return len(res)>0

    def extract_possible_keywords(self,text):
        possible_kw = self.manager.detect_possible_keywords(text)
        keyword,category,best_confidence = possible_kw[0] if len(possible_kw)>0 else (None,None,0)
        if (category == 'negative_page' and best_confidence> 0.7) or (category != 'negative_page' and best_confidence>0.5):
            return {category:[keyword]}
        return {}


    def is_possible_keyword(self,text,check_exact=True):
        if check_exact and self.is_keyword(text):
            return True
        res = self.extract_possible_keywords(text)
        return len(res)>0
