    def partition_tree(self, G: nx.DiGraph, node):
        if G.out_degree(node) == 0 and G.in_degree(node) == 1:
            # TODO if n.path
            data_node = get_tree_data_node(G, node)
            data_node.set_path_string()
            return node
        elif G.in_degree(node) ==1 and G.out_degree(node) == 1:
            child = list(G.successors(node))[0]
            #print(f"Partitioning one offspring {child} of {node}")
            child = self.partition_tree(G, child)
            # replace node with child and remove node from G
            #print(f"Replacing {node} with {child}")

            children = sorted(list(G.successors(node)))
            for child in children:
                #print(f"Partitioning {child} of {node}")
                self.partition_tree(G, child)
            return self.find_partition(G, node)

    def get_DOM_tree(self, tag: bs4.Tag):
        G = nx.DiGraph()
        parent_node = None
        tag.attrs['doc_path'] = ""
        root = SemanticNode(tag, None)
        root.label = self.get_id(root)

    def build_DOM_tree(self, G: nx.DiGraph, node: SemanticNode):
        parent = node.tag
        parent.attrs['doc_path'] +=f">{parent.name}"
        for tag in parent.children:
            if not tag.name:
                continue
            child = SemanticNode(tag, node)
            child.category = self.element_category_helper.get_category(tag)

            child.label = self.get_id(child)
            if child.category[1] == "main":
                self.main_content_node = child.label
            if child.category[1] == "nav":
                continue
            child.path = node.path + [parent.name]
            child.tag.attrs['doc_path'] = ">".join(child.path)
            child_node = child.label

            G.add_node(child_node, data=child)
            G.add_edge(node.label, child_node)
            if tag.name == 'a':
                tag.attrs['doc_path'] +=f">{tag.name}"
                child.link_tag = True
                continue
            self.build_DOM_tree(G, child)

    def prune_DOM_tree(self,G:nx.DiGraph,root):
        if self.main_content_node is None:
            #self.main_content_node = prune_DOM_tree_CNR(G,root)
            if self.main_content_node is None:
                return G,root
        main_descendants = nx.descendants(G, self.main_content_node)
        main_descendants.add(self.main_content_node)
        for node in list(G.nodes):
            if node not in main_descendants:
                G.remove_node(node)
        return G,self.main_content_node


    def get_id(self, node: SemanticNode):

        self.id_map[node.label] = node.id + 1
        class_ = ".".join(node.tag.get("class",""))
        return f"{node.label}_{node.id}"

    def build_DOM_tree(self, G: nx.DiGraph, node: SemanticNode):
        parent = node.tag
        parent.attrs['doc_path'] +=f">{parent.name}"
        for tag in parent.children:
            if not tag.name:
                continue
            child = SemanticNode(tag, node)
            child.category = self.element_category_helper.get_category(tag)

            child.label = self.get_id(child)
            if child.category[1] == "main":
                self.main_content_node = child.label
            if child.category[1] == "nav":
                continue
            child.path = node.path + [parent.name]
            child.tag.attrs['doc_path'] = ">".join(child.path)
            child_node = child.label

            G.add_node(child_node, data=child)
            G.add_edge(node.label, child_node)
            if tag.name == 'a':
                tag.attrs['doc_path'] +=f">{tag.name}"
                child.link_tag = True
                continue
            self.build_DOM_tree(G, child)

                return G,root

        main_descendants.add(self.main_content_node)
        for node in list(G.nodes):
            if node not in main_descendants:
                G.remove_node(node)
        return G,self.main_content_node

def partition_tree(self, G: nx.DiGraph, node):
    if G.out_degree(node) == 0 and G.in_degree(node) == 1:
        # TODO if n.path
        data_node = get_tree_data_node(G, node)
        data_node.set_path_string()
        return node
    elif G.in_degree(node) ==1 and G.out_degree(node) == 1:
        child = list(G.successors(node))[0]
        #print(f"Partitioning one offspring {child} of {node}")
        child = self.partition_tree(G, child)
        # replace node with child and remove node from G
        #print(f"Replacing {node} with {child}")
        self._replace_node(G, node, child)
        return child
    else:
        children = sorted(list(G.successors(node)))
        for child in children:
            #print(f"Partitioning {child} of {node}")
            self.partition_tree(G, child)
        return self.find_partition(G, node)
