from typing import Union, Any

from spacy import Language

from academic_crawler.config.predefined import keywords_map
from academic_crawler.config.singleton import NlpSingleton
from academic_crawler.config.utils import get_noun_phrases_from_text
from academic_crawler.heuristics.page_vocabulary.similarity import VocabSimilarityManager
from academic_crawler.heuristics.page_vocabulary.vocabulary import DomainVocabulary

class OrgVocabularyManager(object):
    nlp: Union[Language, Any]

    def __init__(self) -> None:
        super().__init__()
        self.nlp = NlpSingleton.get_instance(embeddings=True)
        self.vocabulary = DomainVocabulary(self.nlp)
        self.similarity_manager = VocabSimilarityManager(self.vocabulary)
        self.candidates_keyword= set()



    def add_keyword_candidate(self,keyword,category):
        self.candidates_keyword.add((keyword,category))


    def insert_candidates_into_vocabulary(self):
        for keyword,category in self.candidates_keyword:
            self.vocabulary.add_term(keyword,category)
        self.vocabulary.rebuild_vectors()
        self.candidates_keyword = set()


    def detect_possible_keywords(self,text) -> list[tuple[str,str,float]]:
        doc = self.nlp(text)
        result = []
        noun_phrases = get_noun_phrases_from_text(doc)
        for noun_phrase in noun_phrases:
            category,confidence = self.similarity_manager.get_keyword_category(noun_phrase)
            if category:
                self.add_keyword_candidate(noun_phrase,category)
                result.append((noun_phrase,category,confidence))
        result = sorted(result,key=lambda x:x[2],reverse=True)
        return result


    def get_noun_phrases_rating(self,noun_phrases,category=None):
        ratings = []
        for noun_phrase in noun_phrases:
            rating_category = self.similarity_manager.get_categories_similarity(noun_phrase)
            #max_category = self.similarity_manager.get_max_category(rating_category)
            #max_category = category
            if category:
                category_rankings = self.similarity_manager.get_category_top_n_similar(noun_phrase,category)
                top_match = category_rankings[0][1]
                ratings.append((noun_phrase,top_match))
        ratings = sorted(ratings,key=lambda x:x[1],reverse=True)
        return ratings
